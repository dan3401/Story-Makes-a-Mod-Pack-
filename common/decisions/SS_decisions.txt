decision_civilian_to_trade = {
	owned_planets_only = yes

	resources = {
		category = decisions
		cost = {
			influence = 25
		}
	}

	potential = {
		is_planet_class = pc_living_station
		owner = {
			has_origin = origin_living_station
			NOT = {
				is_gestalt = yes
			}
		}
		NOT = {
			has_modifier = trade_focus
		}
	}

	effect = {
		custom_tooltip = decision_civilian_to_trade_effects
		add_modifier = { modifier = "trade_focus" days = -1 }
		remove_modifier = agri_focus
	}

	ai_weight = {
		weight = 9999
	}
}

decision_upgrade = {
	owned_planets_only = yes

	enactment_time = 900
	resources = {
		category = decisions
		cost = {
			influence = 50
			alloys = 2500
			minerals = 3000
			energy = 3000
		}
	}
	
	allow = {
		free_district_slots < 1
	}

	potential = {
		owner = {
			 has_origin = origin_living_station
		}
		is_planet_class = pc_living_station
		num_pops <= 200
	}

	effect = {
		add_deposit = d_upgrade
	}

	ai_weight = {
		weight = 9999
	}
}
