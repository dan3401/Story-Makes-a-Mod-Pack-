namespace = liv

country_event = {
    id = liv.1

    is_triggered_only = yes
    hide_window = yes

    trigger = {
        owner = {
            has_origin = origin_living_station
			owner = { is_regular_empire = yes }
        }
    }
    immediate = {
        random_owned_planet = {
            limit = { is_capital = yes }
            set_planet_size = 15
            remove_all_districts = yes
			remove_all_buildings = yes
            clear_deposits = yes
			add_deposit = d_upgrade_start_regular
			set_planet_entity = {
				entity = "habitat_phase_02_entity"
				graphical_culture = owner
			}
			
			while = {
				count = 14
				create_pop = {
					species = owner_main_species
					ethos = random
				}
			}

		# 8 jobs
            add_district = district_liv_housing
			add_district = district_liv_housing
		
		# 5 jobs
            add_district = district_liv_bureaucratic
		
		# 15 jobs
			add_district = district_liv_civilian
			add_district = district_liv_civilian
			
		# 6 jobs
			add_district = district_liv_maintenance
			
		# 3 jobs
			add_building = building_capital

		# 6 jobs
			add_building = building_research_lab_1
			add_building = building_research_lab_1
        }
    }
}
