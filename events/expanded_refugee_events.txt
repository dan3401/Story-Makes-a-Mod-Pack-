# Starvation kills a pop
country_event = {
	id = refugeeexpanded.1
	title = "refugeeexpanded.1.name"
	desc = "refugeeexpanded.1.desc"
	picture = GFX_evt_dead_city
	show_sound = event_ghost_town
	location = ROOT
	
	trigger = {
		OR = {
			is_country_type = default
			is_country_type = ascended_empire
			is_country_type = lost_empire
		}
		OR = {
			NOT = { resource_stockpile_compare = { resource = food value > 1 } }
			AND = {
				has_edict = food_rationing
				NOT = { resource_stockpile_compare = { resource = food value > 500 } }
			}
		}
		NOT = { has_country_flag = recent_starvation }
		any_owned_planet = {
			num_pops > 3
			any_pop = { is_robot_pop = no is_lithoid = no }
		}
	}
	
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.75
			has_monthly_income = { resource = food value < -5 } 
		}
		modifier = {
			factor = 0.5
			has_monthly_income = { resource = food value < -10 }
		}
		modifier = {
			factor = 0.5
			has_monthly_income = { resource = food value < -15 }
		}
		modifier = {
			factor = 0.5
			has_monthly_income = { resource = food value < -20 }
		}
		modifier = {
			factor = 0.5
			has_monthly_income = { resource = food value < -25 }
		}
	}
	
	option = {
		name = "TERRIBLE"
		trigger = {
			NOR = {
				has_valid_civic = civic_hive_devouring_swarm
				has_valid_civic = civic_machine_terminator
			}
		}
		hidden_effect = {
			set_timed_country_flag = {
				flag = recent_starvation
				days = 180
			}
		}
		random_owned_planet = {
			limit = {
				num_pops > 3
				any_pop = { is_robot_pop = no is_lithoid = no }
			}
			save_event_target_as = the_planet
			random_pop = {
				limit = {
					is_robot_pop = no
					is_lithoid = no
					OR = {
						is_pop_category = specialist
						is_pop_category = worker
						is_pop_category = simple_drone
						is_enslaved = yes
						is_being_purged = yes
					}
				}
				if = {
					limit = {
						is_being_purged = yes
						ROOT = { has_valid_civic = civic_fanatic_purifiers }
					}
					ROOT = {
						add_monthly_resource_mult = {
							resource = unity
							value = 1.5
							min = 5
							max = 100
						}
					}
				}
				kill_pop = yes
			}
			if = {
				limit = { num_pops > 6 }
				random_pop = {
					limit = {
						is_robot_pop = no
						is_lithoid = no
						OR = {
							is_pop_category = specialist
							is_pop_category = worker
							is_pop_category = simple_drone
							is_enslaved = yes
							is_being_purged = yes
						}
					}
					if = {
						limit = {
							is_being_purged = yes
							ROOT = { has_valid_civic = civic_fanatic_purifiers }
						}
						ROOT = {
							add_monthly_resource_mult = {
								resource = unity
								value = 1.5
								min = 5
								max = 100
							}
						}
					}
					kill_pop = yes
				}
			}
		}
	}
	option = {
		name = "refugeeexpanded.1.b"

pop_event = {
	id = refugeeexpanded.2
	hide_window = yes
	
	trigger = {
		is_valid_refugee_pop = yes
		is_robot_pop = no
		is_lithoid = no
		OR = {
			is_being_purged = no # use normal purge refugee instead
			has_purge_type = { type = purge_displacement }
		}
		exists = owner
		owner = {
			OR = {
				is_country_type = default
				is_country_type = ascended_empire
				is_country_type = lost_empire
			}
		}
		owner = {
			OR = {
				resource_stockpile_compare = { resource = food value < 2 }
				AND = {
					NOT = { has_monthly_income = { resource = food value > 0 } }
					resource_stockpile_compare = { resource = food value < 100 }
				}
				AND = {
					has_edict = food_rationing
					has_monthly_income = { resource = food value <= 5 }
					resource_stockpile_compare = { resource = food value < 500 }
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 500
		modifier = {
			factor = 100
			is_valid_migration_pop = yes
		}
		modifier = {
			factor = 5
			planet = {
				OR = {
					has_modifier = penal_colony
					has_modifier = slave_colony
				}
			}
		}
		modifier = {
			factor = 5
			is_enslaved = yes
		}
		modifier = {
			factor = 10
			is_enslaved = yes
			exists = planet
			planet = {
				NOT = { has_modifier = underground_railroad }
			}
		}
		modifier = {
			factor = 5
			NOR = {
				is_pop_category = worker
				is_enslaved = yes
			}
		}
		modifier = {
			factor = 2
			has_trait = trait_conformists
		}
		modifier = {
			factor = 4
			has_trait = trait_obedient
		}
		
		modifier = {
			factor = 0.25
			exists = owner
			owner = {
				has_country_flag = recent_starvation
				OR = {
					resource_stockpile_compare = { resource = food value < 2 }
					AND = {
						NOT = { has_monthly_income = { resource = food value > 0 } }
						resource_stockpile_compare = { resource = food value < 100 }
					}
					AND = {
						has_edict = food_rationing
						has_monthly_income = { resource = food value <= 5 }
						resource_stockpile_compare = { resource = food value < 500 }
					}
				}
			}
		}
		modifier = {
			factor = 40
			exists = owner
			owner = {
				resource_stockpile_compare = { resource = food value > 50 }
			}
		}
	}
	
	immediate = {
		set_pop_flag = pop_escaped_starvation
		if = {
			limit = {
				exists = owner
				owner = { is_ai = no }
			}
			owner = { country_event = { id = refugeeexpanded.3 } }
		}
		save_event_target_as = refugee_pop #Must be saved for refugee_pop_effect to work
		refugee_pop_effect = yes
	}
}

pop_event = {
	id = refugeeexpanded.4
	hide_window = yes
	
	trigger = {
		exists = owner
		owner = {
			OR = {
				is_country_type = default
				is_country_type = ascended_empire
				is_country_type = lost_empire
			}
		}
		is_valid_refugee_pop = yes
		OR = {
			planet = { has_orbital_bombardment = yes }
			planet = { is_occupied_flag = yes }
			has_modifier = pop_recently_conquered
		}
	}
	
	mean_time_to_happen = {
		months = 500
		modifier = {
			factor = 5
			planet = {
				OR = {
					has_modifier = penal_colony
					has_modifier = slave_colony
				}
			}
		}
		modifier = {
			factor = 5
			is_enslaved = yes
		}
		modifier = {
			factor = 10
			is_enslaved = yes
			exists = planet
			planet = {
				NOT = { has_modifier = underground_railroad }
			}
		}
		modifier = {
			factor = 0.75
			happiness < 0.4
			is_enslaved = no
		}
		modifier = {
			factor = 2
			has_trait = trait_conformists
		}
		modifier = {
			factor = 4
			has_trait = trait_obedient
		}
	}
	
	immediate = {
		save_event_target_as = refugee_pop #Must be saved for refugee_pop_effect to work
		if = {
			limit = {
				NOT = { has_modifier = pop_recently_conquered }
				is_enslaved = no
				is_being_purged = no
				exists = owner
				owner = {
					any_owned_planet = {
						habitability = { who = event_target:refugee_pop value >= 0.4 }
						free_housing >= 1
						is_under_colonization = no
						is_controlled_by = owner
						has_orbital_bombardment = no
						is_occupied_flag = no
					}
				}
			}
			random_list = {
				4 = {
					owner = {
						random_owned_planet = {
							limit = {
								habitability = { who = event_target:refugee_pop value >= 0.4 }
								free_housing >= 1
								is_under_colonization = no
								is_controlled_by = owner
								has_orbital_bombardment = no
								is_occupied_flag = no
							}
							resettle_pop = {
								pop = event_target:refugee_pop
								planet = this
							}
						}
					}
				}
				1 = {
					set_pop_flag = pop_escaped_war
					refugee_pop_effect = yes
				}
			}
		}
		else = {
			set_pop_flag = pop_escaped_war
			refugee_pop_effect = yes
		}
	}
}
